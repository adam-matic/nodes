
import euler_integrator

module triangle_wave_generator {
    param period = 20
    param amplitude = 5
    param dt = 0.1

    // Simple discrete triangle wave implementation
    // Direction: 1 for up, -1 for down
    direction = mem(1, next_direction)

    // Current value with memory
    current_value = mem(0, next_value)

    // Calculate next value by adding direction
    next_raw_value = add(current_value, direction)

    // Check if we need to turn around
    will_exceed_positive = gt(next_raw_value, amplitude)
    will_exceed_negative = lt(next_raw_value, -amplitude)

    // When we would exceed limits, flip direction and adjust value
    // If we would go above amplitude, go to amplitude-1 instead
    turn_down_value = sub(amplitude, 1)
    // If we would go below -amplitude, go to -amplitude+1 instead
    turn_up_value = add(-amplitude, 1)

    // Calculate flipped directions
    minus_one = sub(0, 1)
    flipped_direction = mult(direction, minus_one)

    // Determine next direction
    next_direction_if_turn = flipped_direction
    next_direction_if_normal = direction
    should_turn = add(will_exceed_positive, will_exceed_negative)
    one_minus_should_turn = sub(1, should_turn)
    term1 = mult(next_direction_if_turn, should_turn)
    term2 = mult(next_direction_if_normal, one_minus_should_turn)
    next_direction = add(term1, term2)

    // Determine next value
    turn_down_term = mult(turn_down_value, will_exceed_positive)
    turn_up_term = mult(turn_up_value, will_exceed_negative)
    normal_term = mult(next_raw_value, one_minus_should_turn)
    next_value = add(turn_down_term, add(turn_up_term, normal_term))

    output current_value
}

module triangle_wave_example {
    triangle_out = triangle_wave_generator(period=20, amplitude=5)
    output triangle_out
}

execution {
    max_steps: 200
    save: [triangle_out]
}
