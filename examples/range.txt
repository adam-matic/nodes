module range {
    param start
    param step_size
    param stop

    // Check if next value would exceed stop before computing it
    next_step_offset = mult(add($step, 1), step_size)
    next_value = add(start, next_step_offset)
    should_stop = gt(next_value, stop)
    HALT(should_stop)

    step_offset = mult($step, step_size)
    current_value = add(start, step_offset)

    output current_value
}

// This module generates a range of values starting from `start` and
// incrementing by `step_size` each step. The output follows the formula:
// output = start + ($step * step_size)
// The module will halt execution when the current value reaches or exceeds
// the `stop` parameter using the HALT block.
