module min {
    input a
    input b

    is_a_less = lt(a, b)
    term1 = mult(a, is_a_less)
    one_minus_is_a_less = sub(1, is_a_less)
    term2 = mult(b, one_minus_is_a_less)
    output = add(term1, term2)
}

module ramp_generator {
    param ramp_end_value = 5

    // The ramp is the step counter
    ramp = $step

    // The output is the minimum of the ramp and the end value.
    ramped = min(a=ramp, b=ramp_end_value)
    
    output ramped
}

// This example demonstrates a ramp generator that stops at a specific
// value by taking the minimum of the current step and the end value.

execution {
    max_steps: 10
    save: [ramped]
}
