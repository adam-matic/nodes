
import euler_integrator

module sine_wave_generator {
    param frequency = 1
    param amplitude = 1
    param dt = 0.1

    // To generate a sine wave, we can use a second-order system
    // where the acceleration is proportional to the negative of the position.
    // d2x/dt2 = -k * x
    // This can be modeled with two integrators.

    // Integrator 1: integrates velocity to get position (the sine wave)
    // Integrator 2: integrates acceleration to get velocity (the cosine wave)

    // acceleration = -(2πf)^2 * position (proper angular frequency)
    pi = 3.14159265359
    two_pi = mult(2, pi)
    angular_freq = mult(two_pi, frequency)
    angular_freq_squared = mult(angular_freq, angular_freq)
    minus_one = sub(0, 1)
    neg_angular_freq_squared = mult(minus_one, angular_freq_squared)

    // Use memory-based approach to break circular dependency
    position_delayed = mem(0, position)
    acceleration = mult(neg_angular_freq_squared, position_delayed)

    // Manual Euler integration for velocity with proper initial condition
    velocity_prev = mem(0, velocity)
    is_first_step = eq($step, 0)
    initial_velocity_value = mult(angular_freq, amplitude)

    // velocity[n] = velocity[n-1] + acceleration[n-1] * dt
    // but velocity[0] = initial_velocity
    velocity_increment = mult(acceleration, dt)
    velocity_updated = add(velocity_prev, velocity_increment)
    term1 = mult(velocity_updated, sub(1, is_first_step))
    term2 = mult(initial_velocity_value, is_first_step)
    velocity_with_init = add(term1, term2)
    velocity = velocity_with_init

    // position is the integral of velocity
    // Initial position is 0 to start with a sine
    position = euler_integrator(signal=velocity, dt=dt, initial_value=0)

    output position
}

module sine_example {
    sine_out = sine_wave_generator(frequency=0.1, amplitude=1)
    output sine_out
}

execution {
    max_steps: 100
    save: [sine_out]
}
