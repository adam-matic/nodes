module absolute_value {
    input signal

    // is_positive is 1 if signal > 0, else 0
    is_positive = gt(signal, 0)
    // is_not_positive is 1 if signal <= 0, else 0
    is_not_positive = lte(signal, 0)

    // If positive, result is `signal`. If not positive, result is `-signal`.
    // One of the terms in the sum will always be zero.
    abs_signal = (signal * is_positive) + (-signal * is_not_positive)

    output abs_signal
}

// This example demonstrates how to compute the absolute value of a signal.
// We will use a range that goes from -5 to 5.

module abs_example {
    // The input signal goes from -5 to 5
    input_sig = -5 + $step

    abs_output = absolute_value(signal=input_sig)

    output abs_output
}

execution {
    max_steps: 10
    save: [input_sig, abs_output]
}
