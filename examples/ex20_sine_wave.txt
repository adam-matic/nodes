
import euler_integrator

module sine_wave_generator {
    param frequency = 1
    param amplitude = 1
    param dt = 0.1

    // To generate a sine wave, we can use a second-order system
    // where the acceleration is proportional to the negative of the position.
    // d2x/dt2 = -k * x
    // This can be modeled with two integrators.

    // Integrator 1: integrates velocity to get position (the sine wave)
    // Integrator 2: integrates acceleration to get velocity (the cosine wave)

    // acceleration = -frequency^2 * position
    acceleration = -(frequency * frequency) * position

    // velocity is the integral of acceleration
    // Initial velocity is `amplitude * frequency` to start with a cosine
    velocity = euler_integrator(signal=acceleration, dt=dt, initial_value=amplitude * frequency)

    // position is the integral of velocity
    // Initial position is 0 to start with a sine
    position = euler_integrator(signal=velocity, dt=dt, initial_value=0)

    output position
}

module sine_example {
    sine_out = sine_wave_generator(frequency=1, amplitude=5)
    output sine_out
}

execution {
    max_steps: 100
    save: [sine_out]
}
