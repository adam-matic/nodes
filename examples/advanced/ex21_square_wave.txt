
import toggle_switch

module square_wave_generator {
    param period = 10
    param amplitude = 5

    // Use the toggle switch to create a 0-1 signal
    // For a square wave, we want to toggle every period/2 steps
    half_period = div(period, 2)
    toggle = toggle_switch(period=half_period)

    // Scale and shift the 0-1 signal to be a -amplitude to +amplitude square wave
    // Formula: output = (2 * toggle - 1) * amplitude
    two_toggle = mult(2, toggle)
    toggle_minus_one = sub(two_toggle, 1)
    scaled_wave = mult(toggle_minus_one, amplitude)

    output scaled_wave
}

module square_wave_example {
    param period = 20
    param amplitude = 2

    square_out = square_wave_generator(period=period, amplitude=amplitude)
    output square_out
}

execution {
    max_steps: 100
    save: [square_out]
}
