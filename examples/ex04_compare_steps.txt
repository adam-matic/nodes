module compare_steps {
    param threshold = 5

    // Use the `gt` (greater than) comparison block.
    // The output will be 1 if $step > threshold, and 0 otherwise.
    is_over_threshold = gt($step, threshold)

    output is_over_threshold
}

// This example demonstrates the use of comparison blocks.
// The output will switch from 0 to 1 when the step counter
// exceeds the threshold.

execution {
    max_steps: 10
    save: [is_over_threshold, $step]
}
