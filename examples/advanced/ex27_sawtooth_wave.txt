
module sawtooth_wave_generator {
    param period = 20
    param amplitude = 5

    // Internal counter that goes from 0 to period-1 and repeats
    next_count = add(count, 1)
    is_reset = gte(next_count, period)
    one_minus_is_reset = sub(1, is_reset)
    reset_mult = mult(next_count, one_minus_is_reset)
    count = mem(0, reset_mult)

    // The output is the counter value, scaled to the amplitude
    // The slope is amplitude / period
    slope = div(amplitude, period)
    wave = mult(count, slope)

    output wave
}

module sawtooth_wave_example {
    sawtooth_out = sawtooth_wave_generator(period=5, amplitude=5)
    output sawtooth_out
}

execution {
    max_steps: 100
    save: [sawtooth_out]
}
