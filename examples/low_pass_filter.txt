module low_pass_filter {
    input signal
    param alpha = 0.1 // Smoothing factor. Closer to 0 means more smoothing.

    // y_out = y_prev + alpha * (x_in - y_prev)
    // y_out = (1 - alpha) * y_prev + alpha * x_in

    prev_output = mem(0, output)
    one_minus_alpha = sub(1, alpha)
    term1 = mult(one_minus_alpha, prev_output)
    term2 = mult(alpha, signal)
    output = add(term1, term2)

    output output
}
